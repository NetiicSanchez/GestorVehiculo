import { Component, OnInit, ChangeDetectorRef } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { VehiculosService } from '../services/vehiculos.service';
import { CombustibleService } from '../services/combustible.service';
import { Vehiculo } from '../models/vehiculo.model';
import { NgxChartsModule, Color, ScaleType } from '@swimlane/ngx-charts';
import { curveLinear } from 'd3-shape';

@Component({
  selector: 'app-graficas-gastos',
  standalone: true,
  imports: [
    CommonModule,
    FormsModule,
    NgxChartsModule
  ],
  templateUrl: './graficas-gastos.html',
  styleUrls: ['./graficas-gastos.css']
})
export class GraficasGastos implements OnInit {
  errorGastos: string | null = null;
  errorRendimiento: string | null = null;
  // ...existing code...

  trackByVehiculoId(index: number, vehiculo: Vehiculo): number {
  return vehiculo.id ?? index;
  }
  vehiculos: Vehiculo[] = [];
  selectedVehiculoId: number | null = null;

  // Gráfica de Gastos (NGX-CHARTS)
  gastosData: any[] = [];
  gastosColorScheme: Color = {
    name: 'gastosScheme',
    selectable: true,
    group: ScaleType.Ordinal,
    domain: ['#00bfff'] // azul vibrante
  };

  // Gráfica de Rendimiento (NGX-CHARTS)
  // Gráfica de Rendimiento (NGX-CHARTS)
  rendimientoData: any[] = [];
  rendimientoColorScheme: Color = {
    name: 'rendimientoScheme',
    selectable: true,
    group: ScaleType.Ordinal,
    domain: ['#00bfff'] // azul vibrante
  };
  view: [number, number] = [window.innerWidth > 1200 ? 1100 : window.innerWidth - 80, 450];
  showLegend = window.innerWidth > 700;

  onResize(event: any) {
    const width = event.target.innerWidth > 1200 ? 1100 : event.target.innerWidth - 80;
  this.view = [width, 450];
  this.showLegend = event.target.innerWidth > 700;
  }
  showXAxis = true;
  showYAxis = true;
  gradient = false;
  curve = curveLinear;
  timeline = true;
  showXAxisLabel = true;
  showYAxisLabel = true;
  xAxisLabelGastos = 'Fecha de Carga';
  yAxisLabelGastos = 'Costo de Carga';
  xAxisLabelRendimiento = 'Fecha de Carga';
  yAxisLabelRendimiento = 'Rendimiento (Km/Galón)';


  constructor(
    private vehiculosService: VehiculosService,
    private combustibleService: CombustibleService,
    private cdr: ChangeDetectorRef
  ) {}
  ngOnInit(): void {
  window.addEventListener('resize', this.onResize.bind(this));
    this.vehiculosService.obtenerVehiculos().subscribe((response: any) => {
      this.vehiculos = response && response.data ? response.data : [];
    });
  }

  onVehiculoChange(): void {
    this.errorGastos = null;
    this.errorRendimiento = null;
    if (this.selectedVehiculoId) {
      this.combustibleService.getGastosCombustiblePorVehiculo(this.selectedVehiculoId).subscribe({
        next: (response: any) => {
            console.clear();
            console.log('[GASTOS] Datos crudos recibidos:', response);
            let gastos = response && response.data ? response.data : [];
            let data = Array.isArray(gastos)
              ? gastos
                  .map(item => {
                    const fechaValida = item.fecha_carga && String(item.fecha_carga).trim() !== '';
                    // Permitir total_pagado o costo_total (string o número)
                    let valor = null;
                    if (item.total_pagado !== undefined) {
                      valor = Number(item.total_pagado);
                    } else if (item.costo_total !== undefined) {
                      valor = Number(item.costo_total);
                    }
                    if (!fechaValida) {
                      console.warn('[GASTOS] Registro descartado por fecha_carga inválida:', item);
                    }
                    if (!Number.isFinite(valor)) {
                      console.warn('[GASTOS] Registro descartado por total_pagado/costo_total NaN:', item);
                    }
                    let fechaStr = '';
                    if (fechaValida) {
                      // Si ya es string tipo ISO, úsalo; si es Date, conviértelo; si es string con hora, recorta solo la fecha
                      if (typeof item.fecha_carga === 'string') {
                        // Forzar formato YYYY-MM-DD para máxima compatibilidad
                        const match = item.fecha_carga.match(/^\d{4}-\d{2}-\d{2}/);
                        fechaStr = match ? match[0] : item.fecha_carga;
                      } else if (item.fecha_carga instanceof Date) {
                        fechaStr = item.fecha_carga.toISOString().slice(0, 10);
                      } else {
                        // Intentar parsear cualquier otro formato
                        const d = new Date(item.fecha_carga);
                        fechaStr = isNaN(d.getTime()) ? String(item.fecha_carga) : d.toISOString().slice(0, 10);
                      }
                    }
                    return fechaValida && Number.isFinite(valor)
                      ? {
                          name: fechaStr,
                          value: valor
                        }
                      : null;
                  })
                  .filter(point => point !== null)
              : [];
            if (data.length === 0 && Array.isArray(gastos) && gastos.length > 0) {
              console.error('[GASTOS] Todos los registros fueron descartados. Ejemplo de datos crudos:', gastos.slice(0,3));
            }
            if (!data.length) {
              this.errorGastos = 'No hay datos de gastos con fecha válida para este vehículo.';
              this.gastosData = [];
            } else {
              this.gastosData = [{ name: 'Costo de Carga', series: data }];
            }
            console.log('[GASTOS] Datos procesados para ngx-charts:', data);
            this.cdr.detectChanges();
        },
        error: (err) => {
          this.errorGastos = 'Error al obtener los datos de gastos.';
          this.gastosData = [];
        }
      });
      this.combustibleService.getRendimientoVehiculo(this.selectedVehiculoId).subscribe({
        next: (response: any) => {
          console.log('[RENDIMIENTO] Datos crudos recibidos:', response);
          const rendimiento = response && response.data ? response.data : [];
          const data = Array.isArray(rendimiento)
            ? rendimiento.map(item => ({
                name: item.fecha ? (typeof item.fecha === 'string' ? item.fecha : (new Date(item.fecha).toLocaleDateString?.() || String(item.fecha))) : 'Sin fecha',
                value: item.rendimiento !== undefined && item.rendimiento !== null ? Number(item.rendimiento) : 0
              }))
            : [];
          if (!data.length) {
            this.errorRendimiento = 'No hay datos de rendimiento para este vehículo.';
          }
          console.log('[RENDIMIENTO] Datos procesados para ngx-charts:', data);
          this.rendimientoData = [{ name: 'Rendimiento (Km/Galón)', series: data }];
          this.cdr.detectChanges();
        },
        error: (err) => {
          this.errorRendimiento = 'Error al obtener los datos de rendimiento.';
          this.rendimientoData = [];
        }
      });
    }
  }
}
